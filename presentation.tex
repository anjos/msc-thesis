\rottransp{Conclusões}
\small
\begin{enumerate}
\item Discriminação
	\begin{enumerate} 
	\item Discriminadores neuronais são mais eficientes mesmo usando as 4
	variáveis clássicas;
	\item As somas em anel mostraram ser uma forma de compactação que
	aproveita melhor a granularidade do calorímetro;
	\item Estudo de relevância permitiu a compactação dos dados de entrada
	(5 anéis é ainda melhor que as 4 clássicas!);
	\end{enumerate}

\item Implementação
	\begin{enumerate}
	\item Esta foi a primeira vez que o sistema completo foi implementado;
	\item Esta implementação está de acordo com os padrões do RefSoft;
	\item A maior parte do tempo de processamento (87\%) é gasto com a
	aquisição; 
	\item A utilização de LUT's não é necessária;
	\item A implementação em um ADSP 21062 @ 40 MHz torna o processamento
	viável para atuar no L2, rodando mais rápido até que o sistema de
	discriminação clássico (13 $\longrightarrow$ 1 ms);
	\item O discriminador neuronal é mais rápido e eficiente que o sistema
	clássico (18 $\times$ 50 $\mu$s);
	\end{enumerate}
\end{enumerate}
